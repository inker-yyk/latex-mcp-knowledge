\begin&#123;tikzpicture&#125;
  [transform shape,
   every day/.style=&#123;anchor=mid,font=\fontsize&#123;6&#125;&#123;6&#125;\selectfont&#125;]

  &#123;
    % Compute angle:
    \mycount=\month
    \advance\mycount by -1
    \multiply\mycount by 30
    \advance\mycount by -90

      \multiply\mycount by 11
      \advance\mycount by 90

    &#125;;
  &#125;
