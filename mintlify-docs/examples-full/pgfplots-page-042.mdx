      \fi
    \fi
    \expandafter\scope\expandafter[\markopts]
  &#125;,
  scatter/@post marker code/.code=&#123;
    \endscope
  &#125;,
]
    \addplot+ [scatter,scatter src=y,samples=40]
       &#123;sin(deg(x))&#125;;
\end&#123;axis&#125;
