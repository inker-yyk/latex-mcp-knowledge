  int i;
  for (i = 0; i &lt; graph-&gt;vertices.length; i++) &#123;
    pgfgd_Vertex* v = graph-&gt;vertices.array[i];
    v-&gt;pos.x = cos(angle*i) * radius;
    v-&gt;pos.y = sin(angle*i) * radius;
  &#125;
