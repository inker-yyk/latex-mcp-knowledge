    -- Connect the points p and q
    local x1, y1, x2, y2 = math.floor(p.x+0.5), math.floor(p.y+0.5), math.floor(q.x+0.5), math.floor(q.y+0.5)

    local delta_x = x2-x1
    local delta_y = y2-y1
    ...
      local slope = delta_y/delta_x
      for i=x1,x2 do
        local x,y = i, math.floor(y1 + (i-x1)*slope + 0.5)

        if canvas[x][y] == " " then
          canvas[x][y] = '.'
        end
      end
    ...
  end

  -- Iterate over all points on the path from tail to head:
  local p = e.tail.pos

    p = e.tail.pos + e.path[i]

### Description

head{->}body{`\}}} Declare a rule. head should consist of a single symbol, which need not have been declared using || or exist as a default symbol (in fact, the more interesting L-systems depend on using symbols with no corresponding code, to control the ``growth'' of the system). body consists of a string of symbols, which again need not necessarily have any code associated with them.
