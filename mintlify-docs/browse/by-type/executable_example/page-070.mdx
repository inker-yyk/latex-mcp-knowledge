      of colormap={
          whiteblue,
          ocean height,
          target pos max=0,
          sample for=const,
      },
      of colormap={
          gb,
          ocean height,
          target pos min=0.1,
          sample for=const,
      },
  },

    \addplot3[surf] {x*y};
\end{axis}
\end{tikzpicture}

    colorbar style={colormap access=const}
]

\end{tikzpicture}
