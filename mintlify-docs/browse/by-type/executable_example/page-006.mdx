  of colormap/ocean height/.style={
      target pos={-12000,-10000,-6000,-5000,-3000,

          200,500,1000,1100,1200,1500,2000,4000,
          6000,8000},

      of colormap={
          whiteblue,
          ocean height,
          target pos max=0,
          sample for=const,
      },
      of colormap={
          gb,
          ocean height,
          target pos min=0.1,
          sample for=const,
      },
  },
}

    \addplot3[surf] {x*y};
\end{axis}
\end{tikzpicture}

    colorbar style={colormap access=const}
]
